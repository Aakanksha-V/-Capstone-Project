CREATE SCHEMA `complaintsystem` ;
CREATE TABLE `complaintsystem`.`complaint` (
  `userId` INT NOT NULL,
  `name` VARCHAR(45) NULL,
  `address` VARCHAR(100) NULL,
  `pinCode` VARCHAR(6) NULL,
  `status` VARCHAR(45) NULL,
  `issue` VARCHAR(45) NULL,
  PRIMARY KEY (`userId`));

CREATE TABLE `complaintsystem`.`manager` (
  `areaId` INT NOT NULL,
  `pinCode` VARCHAR(45) NULL,
  `managerId` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`areaId`, `managerId`));
